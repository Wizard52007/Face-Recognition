// ========================================== // THE CORE MEMORY (Stealth / Sequential Puzzle) // ========================================== window.coreMemoryDiary = new THREE.Group(); window.coreMemoryDiary.name = "Room1_Diary"; // REMOVED from interactableObjects so she can't click it early! const bookGeo = new THREE.BoxGeometry(0.6, 1.0, 0.1); const coverTexture = new THREE.TextureLoader().load('diary_cover.jpg'); // 1. THE CAMOUFLAGE: Darken the materials so it blends into the shelf/floor const coverMat = new THREE.MeshStandardMaterial({ map: coverTexture, roughness: 0.9, // Dusty and matte color: 0x444444 // Tints the photo dark grey/black }); const paperMat = new THREE.MeshStandardMaterial({ color: 0x333333, // Dark, aged pages instead of bright white roughness: 1.0 }); const bookMaterials = [paperMat, coverMat, paperMat, paperMat, coverMat, coverMat]; const customDiary = new THREE.Mesh(bookGeo, bookMaterials); customDiary.position.set(6.25, 2.4, 6.85); customDiary.rotation.set(0, -Math.PI / 6, 0); window.coreMemoryDiary.add(customDiary); // 2. THE MAGICAL GLOW (Starts turned OFF) // Notice the second number is 0 (Intensity) window.diaryLight = new THREE.PointLight(0x4444ff, 0, 3); window.diaryLight.position.set(0, 0.5, 0.5); window.coreMemoryDiary.add(window.diaryLight); const diaryHitBoxGeo = new THREE.SphereGeometry(1.0, 16, 16); const diaryHitBoxMat = new THREE.MeshBasicMaterial({ visible: false }); const diaryHitBox = new THREE.Mesh(diaryHitBoxGeo, diaryHitBoxMat); window.coreMemoryDiary.add(diaryHitBox); // 3. INVISIBILITY CLOAK window.coreMemoryDiary.visible = false; scene.add(window.coreMemoryDiary);
